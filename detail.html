// ฟังก์ชัน smartFetch สำหรับดึงข้อมูลรายตลาดที่ยืดหยุ่นขึ้น async function smartFetchRaw(marketName) { const variants = [ `data/raw_${marketName.toUpperCase()}.json`, `data/raw_${marketName.toLowerCase()}.json` ]; for (let path of variants) { try { const res = await fetch(path); if (res.ok) return await res.json(); } catch (e) { continue; } } throw new Error("ไม่พบไฟล์ข้อมูลดิบสำหรับตลาด " + marketName); } // เพิ่มส่วนการแสดง Win-Rate แยก 30/60/90 วันในหน้ารายละเอียด function renderMarketDetail(period) { const stats = marketData.history[period]; const html = `
Net Profit (${period}) ${stats.profit} ฿
Win-Rate ${stats.win_rate}%
`; document.getElementById('stats-display').innerHTML = html; }
